// High-performance RPC client with connection pooling
// and automatic retry logic for blockchain operations.
pub async fn send_transaction(
    &self,
    tx: Transaction,
) -> Result<TxHash, RpcError> {
    // 1. Validate transaction
    self.validator.validate(&tx)?;
    
    // 2. Get optimized gas price
    let gas_price = self
        .gas_oracle
        .get_optimized_price()
        .await?;
    
    // 3. Sign & broadcast
    let signed = self.signer.sign(tx, gas_price)?;
    self.pool.broadcast(signed).await
}
